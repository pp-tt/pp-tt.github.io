    def __init__(self, *args, **kwargs):
        super().__init__(*args, **kwargs)
        self.f1 = Flatten()
        self.d1 = Dense(120, activation='relu')
        self.d2 = Dense(10, activation='softmax')

    def call(self, x):
        x = self.f1(x)
        x = self.d1(x)
        x = self.d2(x)
        return x
