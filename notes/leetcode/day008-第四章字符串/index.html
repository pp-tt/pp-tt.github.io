    class Solution {
    public String reverseWords(String s) {
StringBuffer res = new StringBuffer();

        int endIndex = s.length()-1;
        int startS = 0;

        while (startS <= endIndex && s.charAt(startS) == ' ') {
            startS++;
        }

        while (endIndex >= startS) {

            while (endIndex > startS && s.charAt(endIndex) == ' ') {
                endIndex--;
            }

            int start = endIndex - 1;


            while (start >= startS && s.charAt(start) != ' ') {
                start--;
            }

            res.append(s.substring(start+1, endIndex+1));

            endIndex = start;

            if (start > startS)
                res.append(' ');

        }
        return res.toString();
    }
}
