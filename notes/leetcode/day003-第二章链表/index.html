    class Solution {
    public ListNode reverseList(ListNode head) {
        ListNode dummy = new ListNode(-1, head);
        ListNode p = dummy;
        ListNode newNode = new ListNode();

        while (p.next != null) {
            ListNode node = new ListNode(p.next.val);
            node.next = newNode.next;
            newNode.next = node;
            p = p.next;
        }
        return newNode.next;
    }
