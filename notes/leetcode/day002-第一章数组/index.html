    public int[] sortedSquares(int[] nums) {
        int index = nums.length-1;
        int[] res = new int[index+1];
        int l = 0, r = nums.length-1;
        while (l <= r) {
            if (nums[l] * nums[l] > nums[r] * nums[r] ) res[index--] = nums[l] * nums[l++];
            else res[index--] = nums[r] * nums[r--];
        }
        return res;
    }

    public int[][] generateMatrix(int n) {

        int start = 0;   // èµ·å§‹ä½ç½®

        while ( loop < n / 2 ){

            while (j < n - offset) res[i][j++] = count++;
            while (i < n - offset) res[i++][j] = count++;
            while (j > start) res[i][j--] = count++;
