        public static boolean isAnagram(String s, String t) {
        int[] record = new int[26];
        for (int i = 0; i < s.length(); i++) {
            record[s.charAt(i) - 'a']++;
        }
        for (int i = 0; i < t.length(); i++) {
            record[t.charAt(i) - 'a']--;
        }

                return false;
            }
        }
        return true;
    }

        public static int[] intersection(int[] nums1, int[] nums2) {

                record[num] = -1;
                k++;
            }
        }

        int[] res = new int[k];
        int index = 0;

        for (int num : nums2) {

                res[index++] = num;
                record[num] = 1;
            }
        }

        public static boolean isHappy(int n) {
        Set<Integer> nums = new HashSet<>();
        int res = squreSum(n);
        while (!nums.contains(res)) {
            if (res == 1) {
                return true;
            } else {
                nums.add(res);
            }
            res = squreSum(res);
        }
        return false;

		public static int squreSum(int n) {
        int res = 0;
        while (n != 0) {
            int temp = n % 10;
            res += temp * temp;
            n = (n - temp) / 10;
        }
        return res;
    }

    public static int[] twoSum(int[] nums, int target) {
        int[] res = new int[2];
        for (int i = 0; i < nums.length; i++) {
            int searchNum = target - nums[i];
            for (int j = i+1; j < nums.length; j++) {
                if (searchNum == nums[j]) {
                    res[0] = i;
                    res[1] = j;
                    break;
                }
            }
            if (res[0] + res[1] == target) {
                break;
            }
        }
        return res;
    }
